async function joinCFC() {
try {
let response = await fetch('/join-us', {
method: 'POST',
body: JSON.stringify({
username: '', // enter your name
email: '', // enter your email
QQ: '', // enter your qq
})
if(response.status !== 201) {
throw new Error('Failed to submit POST request')
} else {
console.log('succeed!')
}
} catch (error) {
console.error(error)
}
}

joinCFC()
